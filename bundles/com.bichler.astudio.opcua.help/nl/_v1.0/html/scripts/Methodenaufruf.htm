Methodenaufruf registrieren

Um einen Aufruf einer Methodenimplementierung zu bekommen, ist es notwendig,

diese mit Hilfe des Browsnames am Benachrichtigungssystem zu registrieren.

 

 

load("nashorn:mozilla_compat.js");

importPackage(org.opcfoundation.ua.builtintypes);

importPackage(java.lang);

importPackage(com.hbsoft.comdrv);

 

/**

 * Write listener callback object.

 * Here we implemented listener member function.

 */

function CallMethodListener() {

      

       /**

        * Calls the required method and returns the result

        *

        * @param (NodeId)  context       Operation context of method call.

        * @param (NodeId)  nodeId NodeId of object to which the method  

 *                                belongs to.

        * @param (NodeId)  methodId      NodeId of the method to call.

        * @param (Variant []) inputargs  List with all required input arguments.

        * @return (CallMethodResult)     The result of the execution of the method.

        */

       this.callMethod = function(context, parentId, methodId, inputargs) {

             System.out.println("method called: " + parentId + " : " + methodId);

       }

}

 

/**

 * create javascript method handler,

 * which function will be called from comet driver manager.

 */

var methoddrv = new CallMethodListener();

 

/**

 * create the java mapping of the write listener.

 */

var methodListener = new IMethodListener(methoddrv);

 

/**

 * Register listener to driver manager for a defined browsepath, pointing to the method object

 */

ComDRVManager.getDRVManager().addMethodListener('0:caller', methodListener);
